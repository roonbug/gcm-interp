function fibonacci(n) {
  let sequence = [0, 1]; // Starting sequence with first two terms

  if (n <= 2) {
    return sequence.slice(0, n);
  }

  for (let i = 2; i < n; i++) {
    let nextTerm = sequence[i - 1] + sequence[i - 2];
    sequence.push(nextTerm);
  }

  return sequence;
}
